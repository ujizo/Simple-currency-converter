cmake_minimum_required(VERSION 3.10)
project(SimpleCurrencyConverter)

set(CMAKE_CXX_STANDARD 11)


find_package(CURL REQUIRED)


add_executable(SimpleCurrencyConverter
    main.cpp
    converter.cpp
    api.cpp
    cache.cpp
)


target_link_libraries(SimpleCurrencyConverter PRIVATE ${CURL_LIBRARIES})
target_include_directories(SimpleCurrencyConverter PRIVATE ${CURL_INCLUDE_DIRS})
